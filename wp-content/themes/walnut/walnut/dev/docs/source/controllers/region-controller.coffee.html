    show: (view, options = {}) ->
      _.defaults options,
        loading: false
        region: @region

      @_setMainView view
      @_manageView view, options

    
    _setMainView: (view) ->

      return if @_mainView
      @_mainView = view
      @listenTo view, "close", @close

    
    _manageView: (view, options) ->
      if options.loading
        App.commands.execute "show:loading", view, options
      else
        options.region.show view
